Goal

Security rules (strict; use custom claims for role)
BEGIN_FIRESTORE_RULES
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function isAdmin() { return signedIn() && request.auth.token.role == 'admin'; }
    function isProvider() { return signedIn() && request.auth.token.role in ['provider','admin']; }
    function isSelf(uid) { return signedIn() && request.auth.uid == uid; }

    match /users/{uid} {
      allow create: if signedIn() && isSelf(uid);
      allow read: if signedIn();
      allow update, delete: if isSelf(uid) || isAdmin();
    }

    match /societies/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /categories/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /items/{id} {
      allow read: if true;
      allow create: if signedIn() && request.resource.data.ownerId == request.auth.uid;
      allow update, delete: if signedIn() && resource.data.ownerId == request.auth.uid || isAdmin();
    }

    match /bookings/{id} {
      allow read: if signedIn() &&
        (request.auth.uid in [resource.data.borrowerId, resource.data.lenderId] || isAdmin());
      allow create: if signedIn() && request.resource.data.borrowerId == request.auth.uid;
      allow update: if signedIn() &&
        (request.auth.uid in [resource.data.borrowerId, resource.data.lenderId] || isAdmin());
      allow delete: if isAdmin();
    }

    match /services/{id} {
      allow read: if true;
      allow create: if signedIn() && request.resource.data.providerId == request.auth.uid;
      allow update, delete: if signedIn() && resource.data.providerId == request.auth.uid || isAdmin();
    }

    match /serviceBookings/{id} {
      allow read: if signedIn() &&
        (request.auth.uid in [resource.data.customerId, resource.data.providerId] || isAdmin());
      allow create: if signedIn() && request.resource.data.customerId == request.auth.uid;
      allow update: if signedIn() &&
        (request.auth.uid in [resource.data.customerId, resource.data.providerId] || isAdmin());
      allow delete: if isAdmin();
    }

    match /chatRooms/{roomId} {
      allow read, update, delete: if signedIn() && request.auth.uid in resource.data.participants;
      allow create: if signedIn() && request.resource.data.participants.hasAny([request.auth.uid]);
      match /messages/{messageId} {
        allow read, create: if signedIn() &&
          request.auth.uid in get(/databases/$(database)/documents/chatRooms/$(roomId)).data.participants;
        allow delete: if false;
      }
    }

    match /ratings/{id} {
      allow read: if true;
      allow create: if signedIn() && request.resource.data.fromUserId == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /payments/{id} {
      allow read: if signedIn() &&
        (request.auth.uid in [resource.data.payerId, resource.data.payeeId] || isAdmin());
      allow create, update: if isAdmin(); // written via Functions only
      allow delete: if isAdmin();
    }

    match /notifications/{id} {
      allow read: if signedIn() && resource.data.userId == request.auth.uid;
      allow create, update, delete: if isAdmin();
    }

    match /favorites/{id} {
      allow read, create, delete: if signedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if false;
    }

    match /reports/{id} {
      allow read: if isAdmin();
      allow create: if signedIn() && request.resource.data.reporterId == request.auth.uid;
      allow update: if isAdmin();
      allow delete: if false;
    }
  }
}
END_FIRESTORE_RULES

Cloud Storage rules
BEGIN_STORAGE_RULES
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    function signedIn() { return request.auth != null; }
    function isOwner(uid) { return signedIn() && request.auth.uid == uid; }

    match /item_images/{itemId}/{file} {
      allow read: if true;
      allow write: if signedIn(); // ownership checked in Firestore on publish
    }
    match /service_images/{serviceId}/{file} {
      allow read: if true;
      allow write: if signedIn();
    }
    match /user_docs/{uid}/kyc/{file} {
      allow read, write: if isOwner(uid);
    }
    match /chat_attachments/{roomId}/{file} {
      allow read, write: if signedIn(); // membership verified by app before upload
    }
  }
}
END_STORAGE_RULES

Cloud Functions (implement and wire to buttons/flows)
- onUserCreate: initialize profile, set default role=user, copy phone; add welcome notification.
- geoIndex: store geoHash for items/services using location; helper for nearby queries; use geofire-common.
- createBooking(item): HTTP callable “bookItem”
  - Validate logged-in, item active, dates not overlapping with existing bookings → compute days, baseAmount, deposit, platformFee (Remote Config, default 10%), total.
  - Create booking doc (status=requested, paymentStatus=pending), create payments doc (escrow=true), return payment intent (UPI intent URL deep link).
- createServiceBooking(service): HTTP callable “bookService” similarly.
- onPaymentWebhook/update (provider-agnostic; mock allowed): mark payments.status, set booking.paymentStatus, transition booking.status (accepted → ongoing → completed); release escrow via bookkeeping records.
- onBookingStatusChange/onServiceBookingStatusChange: maintain availability, send FCM to counterparty.
- imageModeration: run Vision/Gemini safety on new images (Storage trigger); if unsafe, mark item/service “paused” and notify owner.
- scheduledCleanup (CRON): auto-cancel requested bookings older than 24h with no response.

Payments (UPI-first, India)

Definition of Done
- A new user can sign up with phone, browse nearby items/services, create a listing, start a booking, and send/receive chat messages.
- Owners/providers can accept/cancel bookings and get notifications.
- Payments flow is integrated (UPI intent or mock), and booking states update accordingly.

Enable Firestore offline persistence (default on Android/iOS). Use “cache‑then‑server” for list views so users see cached results instantly.
Use a “summary doc pattern” for lists: keep only small fields in items/services used on cards (title, price, rating, city, thumbUrl, updatedAt, geoHash). Heavy fields (long description, large photo set) load on the detail screen only. Thumbnails are generated by a Storage trigger and compressed to WebP.
Query shape: where('status','==','active').orderBy('updatedAt','desc').limit(20) with infinite scroll via startAfter(lastDoc). For nearby search, use geohash bounding boxes and composite indexes.
Denormalize to avoid N+1 reads on lists: store ownerName, ratingAvg, ratingCount on the item/service; keep them in sync with Functions.
Reduce chatter: do not open chat listeners until a room is opened; paginate messages. Prefer one listener per visible list, not per card.
Remote Config controls: listPageSize, maxRadiusKm, imageQuality, and feature flags for experiments.
Analytics monitors list latency and cache hit rate; Performance Monitoring tracks payload sizes.
